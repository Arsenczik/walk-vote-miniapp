async function loadStatistics() { let content = document.getElementById('statsContent'); content.innerHTML = '
🏜️

Загрузка лидерборда...

'; try { let [profileRes, leaderboardRes] = await Promise.all([ fetch('/api/profile/' + USER_ID), fetch('/api/leaderboard') ]); let profile = await profileRes.json(); let leaderboard = await leaderboardRes.json(); let totalVibes = profile.vibes || (profile.stats?.created || 0) * 25 + (profile.stats?.attended || 0) * 15; let html = `
🐪
${totalVibes}
🐪
Твои очки
🏆 ЛИДЕРБОРД
${leaderboard.slice(0, 15).map((u, i) => { const isMe = u.user && (u.user.id === USER_ID || u.user_id === USER_ID); const name = escapeHtml(u.user?.name || u.name || 'Караванщик'); const emoji = u.user?.emoji || u.emoji || '🐪'; const score = u.score || u.vibes || 0; return `
#${i+1}
${emoji}
${name}
${score}
`; }).join('')}
${leaderboard.length > 15 ? `
... и ещё ${leaderboard.length - 15} караванщиков
` : ''}
`; content.innerHTML = html; } catch(e) { console.error(e); content.innerHTML = `
🐪

Не удалось загрузить лидерборд

`; } }
